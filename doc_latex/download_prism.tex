\inputencoding{utf8}
\HeaderA{download\_prism}{Download PRISM Climate Data maps}{download.Rul.prism}
%
\begin{Description}
Download PRISM Climate Data maps
\end{Description}
%
\begin{Usage}
\begin{verbatim}
download_prism(
  sp_res = "4km",
  data = "ppt",
  start_date = as.Date("2017-04-01"),
  end_date = as.Date("2017-08-03"),
  t_res = "monthly",
  out_dir = paste0(getwd()),
  pause = 1
)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{sp\_res}] a character vector that denotes the spatial resolution of the
climate maps wanted

\item[\code{data}] a character vector that is one of the following:
ppt, tmin, tmax, tmean, tdmean, vpdmin, or vpdmax

\item[\code{start\_date}] a character vector or date object in the format YYYY-MM-DD

\item[\code{end\_date}] a character vector or date object in the format YYYY-MM-DD

\item[\code{t\_res}] a character vector that denotes the time resolution
of the map. (daily, monthly, or yearly)

\item[\code{out\_dir}] out\_dir is the name of a output directory or folder that will
be created in the working directory. The User can input a name for the
in which the data will be saved in.

\item[\code{pause}] an integer value that denotes the amount of seconds to pause
between each download
\end{ldescription}
\end{Arguments}
%
\begin{Details}
This function creates a folder and downloads PRISM data
for the selected date(s), map type(s), and data map that the user provides.
The downloaded data is stored in the permanent folder.
\end{Details}
