\HeaderA{quebec\_farms}{Quebec, Canada Maple Farm Addresses}{quebec.Rul.farms}
\keyword{datasets}{quebec\_farms}
%
\begin{Description}
Maple syrup farm addresses scraped from
\Rhref{https://www.bonjourquebec.com/en-us/to-see-and-do/delicious-discoveries/sugar-shacks}{bonjourquebec.com}.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
quebec_farms
\end{verbatim}
\end{Usage}
%
\begin{Format}
%
\begin{SubSection}{\code{quebec\_farms}}

A dataframe with 85 rows and 4 columns:
\begin{description}

\item[farm] Farm name
\item[address] Farm address
\item[state] Province
\item[region] Province region

\end{description}

\end{SubSection}

\end{Format}
