\HeaderA{us\_farms}{U.S. Maple Farm Addresses}{us.Rul.farms}
\keyword{datasets}{us\_farms}
%
\begin{Description}
Maple syrup farm addresses scraped from
\Rhref{https://www.maplesyrupfarms.org/}{maplesyrupfarms.org}. Since scraping
the farm names was complicated, a few farm names are cut off at the first
hyphen.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
us_farms
\end{verbatim}
\end{Usage}
%
\begin{Format}
%
\begin{SubSection}{\code{us\_farms}}

A dataframe with 374 rows and 4 columns:
\begin{description}

\item[farm] Farm name
\item[address] Farm address
\item[state] U.S. state
\item[region] U.S. state region

\end{description}

\end{SubSection}

\end{Format}
