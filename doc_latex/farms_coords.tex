\HeaderA{farms\_coords}{North America Maple Farm Addresses and Coordinates}{farms.Rul.coords}
\keyword{datasets}{farms\_coords}
%
\begin{Description}
Maple syrup farm addresses and coordinates scraped from
\Rhref{https://www.maplesyrupfarms.org/}{maplesyrupfarms.org} and
\Rhref{https://www.bonjourquebec.com/en-us/to-see-and-do/delicious-discoveries/sugar-shacks}{bonjourquebec.com}.
Since scraping the farm names was complicated, a few farm names are cut
off at the first hyphen. Farm addresses were geocoded with
\code{\LinkA{ggmap::geocode()}{ggmap::geocode()}}.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
farms_coords
\end{verbatim}
\end{Usage}
%
\begin{Format}
%
\begin{SubSection}{\code{farms\_coords}}

A dataframe with 444 rows and 6 columns:
\begin{description}

\item[farm] Farm name
\item[address] Farm address
\item[state] U.S. state/Canada province
\item[region] U.S. state/Canada province region
\item[lat] Latitude
\item[lon] Longitude

\end{description}

\end{SubSection}

\end{Format}
