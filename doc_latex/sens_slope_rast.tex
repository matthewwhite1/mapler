\HeaderA{sens\_slope\_rast}{Calculate the Sen's slope for every pixel of a raster}{sens.Rul.slope.Rul.rast}
%
\begin{Description}
Calculate the Sen's slope for every pixel of a raster
\end{Description}
%
\begin{Usage}
\begin{verbatim}
sens_slope_rast(t_rast, cores = 1)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{t\_rast}] terra raster stack

\item[\code{cores}] integer amount of cores for parallelization
\end{ldescription}
\end{Arguments}
%
\begin{Value}
terra raster stack with six layers: Sen's slope estimate,
z statistic, p-value, sample size, confidence interval lower bound, and
confidence interval upper bound
\end{Value}
%
\begin{Examples}
\begin{ExampleCode}
test_loca_file <- system.file("extdata", "test_loca_sap_day.tif",
                              package = "mapler")
sap_prop <- terra::rast(test_loca_file)
sens_slope_rast(sap_prop)
\end{ExampleCode}
\end{Examples}
