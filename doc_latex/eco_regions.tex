\HeaderA{eco\_regions}{Level III North America ecoregions}{eco.Rul.regions}
\keyword{datasets}{eco\_regions}
%
\begin{Description}
Level III North America ecoregions sf dataframe pulled from
\Rhref{https://www.epa.gov/eco-research/ecoregions-north-america}{epa.gov}. This
is simplified to a 5km resolution.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
eco_regions
\end{verbatim}
\end{Usage}
%
\begin{Format}
%
\begin{SubSection}{\code{eco\_regions}}

An sf dataframe with 2548 rows and 12 columns:
\begin{description}

\item[NA\_L3CODE] Level III region code
\item[NA\_L3NAME] Level III region name
\item[NA\_L2CODE] Level II region code
\item[NA\_L2NAME] Level II region name
\item[NA\_L1CODE] Level I region code
\item[NA\_L1NAME] Level I region name
\item[NA\_L3KEY] Level III combination of region code and name
\item[NA\_L2KEY] Level II combination of region code and name
\item[NA\_L1KEY] Level I combination of region code and name
\item[Shape\_Leng] Length of region
\item[Shape\_Area] Area of region
\item[geometry] sf geometry polygons

\end{description}

\end{SubSection}

\end{Format}
