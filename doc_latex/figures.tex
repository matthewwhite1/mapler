\HeaderA{get\_ghcnd\_country\_ids}{Get all GHCNd station ids for a given country}{get.Rul.ghcnd.Rul.country.Rul.ids}
%
\begin{Description}
Get all GHCNd station ids for a given country
\end{Description}
%
\begin{Usage}
\begin{verbatim}
get_ghcnd_country_ids(country_code)
\end{verbatim}
\end{Usage}
%
\begin{Arguments}
\begin{ldescription}
\item[\code{country\_code}] character vector of WMO country codes
\end{ldescription}
\end{Arguments}
%
\begin{Value}
character vector of station ids
\end{Value}
%
\begin{Examples}
\begin{ExampleCode}
# Get all U.S. country ids
get_ghcnd_country_ids("US")
\end{ExampleCode}
\end{Examples}
