\HeaderA{canada\_farms}{Canada Maple Farm Addresses}{canada.Rul.farms}
\keyword{datasets}{canada\_farms}
%
\begin{Description}
Maple syrup farm addresses scraped from
\Rhref{https://www.maplesyrupfarms.org/}{maplesyrupfarms.org}.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
canada_farms
\end{verbatim}
\end{Usage}
%
\begin{Format}
%
\begin{SubSection}{\code{canada\_farms}}

A dataframe with 15 rows and 4 columns:
\begin{description}

\item[farm] Farm name
\item[address] Farm address
\item[state] Province
\item[region] Province region

\end{description}

\end{SubSection}

\end{Format}
